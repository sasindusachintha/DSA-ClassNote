✅ Singly Linked List – Student Marks Management
public class StudentLinkedList {

    // Node class
    class Node {
        int id;
        String name;
        double marks;
        Node next;

        Node(int id, String name, double marks) {
            this.id = id;
            this.name = name;
            this.marks = marks;
            this.next = null;
        }
    }

    Node head = null;

    // 1️⃣ Insert at rear
    public void insertAtRear(int id, String name, double marks) {
        Node newNode = new Node(id, name, marks);

        if (head == null) {
            head = newNode;
        } else {
            Node temp = head;
            while (temp.next != null) {
                temp = temp.next;
            }
            temp.next = newNode;
        }
    }

    // 2️⃣ Insert at front
    public void insertAtFront(int id, String name, double marks) {
        Node newNode = new Node(id, name, marks);
        newNode.next = head;
        head = newNode;
    }

    // 3️⃣ Insert at specific position
    public void insertAtPosition(int id, String name, double marks, int position) {
        Node newNode = new Node(id, name, marks);

        if (position == 1) {
            insertAtFront(id, name, marks);
            return;
        }

        Node temp = head;
        for (int i = 1; i < position - 1 && temp != null; i++) {
            temp = temp.next;
        }

        if (temp == null) {
            System.out.println("Invalid Position!");
            return;
        }

        newNode.next = temp.next;
        temp.next = newNode;
    }

    // 4️⃣ Delete rear node
    public void deleteRear() {
        if (head == null) {
            System.out.println("List is empty.");
            return;
        }

        if (head.next == null) {
            head = null;
            return;
        }

        Node temp = head;
        while (temp.next.next != null) {
            temp = temp.next;
        }

        temp.next = null;
    }

    // 5️⃣ Search by student ID
    public void search(int id) {
        Node temp = head;

        while (temp != null) {
            if (temp.id == id) {
                System.out.println("Student Found:");
                System.out.println("ID: " + temp.id +
                                   ", Name: " + temp.name +
                                   ", Marks: " + temp.marks);
                return;
            }
            temp = temp.next;
        }

        System.out.println("Student not found.");
    }

    // 6️⃣ Find student with highest marks
    public void highestMarks() {
        if (head == null) {
            System.out.println("List is empty.");
            return;
        }

        Node temp = head;
        Node maxNode = head;

        while (temp != null) {
            if (temp.marks > maxNode.marks) {
                maxNode = temp;
            }
            temp = temp.next;
        }

        System.out.println("Student with Highest Marks:");
        System.out.println("ID: " + maxNode.id +
                           ", Name: " + maxNode.name +
                           ", Marks: " + maxNode.marks);
    }

    // Print list
    public void printList() {
        Node temp = head;
        while (temp != null) {
            System.out.println("ID: " + temp.id +
                               ", Name: " + temp.name +
                               ", Marks: " + temp.marks);
            temp = temp.next;
        }
    }

    // Main method
    public static void main(String[] args) {
        StudentLinkedList list = new StudentLinkedList();

        // Insert at rear
        list.insertAtRear(1, "Ali", 75.5);
        list.insertAtRear(2, "Sara", 82.0);

        // Insert at front
        list.insertAtFront(3, "John", 90.0);

        // Insert 5 records at positions 2,3,4,5,6
        list.insertAtPosition(4, "Emma", 88.0, 2);
        list.insertAtPosition(5, "Mike", 70.0, 3);
        list.insertAtPosition(6, "Nina", 95.0, 4);
        list.insertAtPosition(7, "Tom", 60.0, 5);
        list.insertAtPosition(8, "Lily", 85.0, 6);

        // Delete rear
        list.deleteRear();

        // Search
        list.search(6);

        // Highest marks
        list.highestMarks();

        // Print all
        System.out.println("\nAll Students:");
        list.printList();
    }
}
